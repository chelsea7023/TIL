package swea_6190_정곤이의단조증가하는수;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class Solution {
	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);

		int T = sc.nextInt();

		for (int tc = 1; tc <= T; tc++) {

			int N = sc.nextInt();

			int[] num = new int[N];
			for (int i = 0; i < N; i++) {
				num[i] = sc.nextInt();
			}
			// 여기까지 값을 넣은 배열 완성
			// 이제 값들을 단조 배열인지 아닌지를 알아내고 아니면 빼고 맞으면 그대로 둔 상태에서 내림차순으로 정렬

			// 단조 값들을 곱한 값들을 저장할 배열 생성

			List<Integer> numbers = new ArrayList<>();

			for (int i = 0; i < N; i++) {
				for (int j = i + 1; j < N; j++) {
					numbers.add(num[i] * num[j]);
				}
			}
			Collections.sort(numbers, Collections.reverseOrder());

			System.out.println(danjo(12345321));
			
		}
	}

	// 단조 숫자인지 아닌지 확인하는 방법
	static boolean danjo(int a) {

		
		boolean result = true;
		
		for (int i = 1; i < Math.pow(10, 8); i *= 10) {
			if (a < 10) {
				result = true;
				break;
			} else if (!((a / i) % 10 >= (a / (i * 10)) % 10)) {
				result = false;
				break;
			}
		}
		return result;
	
	}
}
