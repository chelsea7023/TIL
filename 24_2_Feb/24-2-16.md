package swea_6190_정곤이의단조증가하는수;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class Solution {
	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		
		int T = sc.nextInt();
		
		for(int tc = 1; tc <= T; tc++) {
			
			int N = sc.nextInt();
			
			int[] num = new int[N];
			for(int i = 0; i < N; i++) {
				num[i] = sc.nextInt();
			}
			// 여기까지 값을 넣은 배열 완성 
			// 이제 값들을 단조 배열인지 아닌지를 알아내고 아니면 빼고 맞으면 그대로 둔 상태에서 내림차순으로 정렬
			
			
			//단조 값들을 곱한 값들을 저장할 배열 생성
			
			List<Integer> numbers = new ArrayList<>();
			
			for(int i = 0; i < N; i++) {
				for(int j = i+1; j < N; j++) {
					numbers.add(num[i]*num[j]);
				}
			}
			Collections.sort(numbers, Collections.reverseOrder());
//			System.out.println(numbers);
			
			int result = -1; 
			
			for(int i = 0; i < numbers.size(); i++) {
				if(danjo(numbers.get(i))) {
					result = numbers.get(i);
					System.out.println("#" + tc + " " + result);
					break;
				}
				continue;
			}
			
			if(result == -1) {
				System.out.println("#" + tc + " " + result);
			}
			
			
//	
		}
	}
	// 단조 숫자인지 아닌지 확인하는 방법
	static boolean danjo(int a) {
		if(a <= 99999 &&a >= 10000) {
			if(a/10000 <= a/1000 && a/1000 <= a/100 && a/100 <= a/10 && a/10 <= a%10) {
				return true;
			}
		}
		else if (a <= 9999 && a >= 1000) {
			if (a/1000 <= a/100 && a/100 <= a/10 && a/10 <= a%10) {
				return true;
			}
				
		}
		else if(a <= 999 && a >= 100) {
			if(a/100 <= a/10 && a/10 <= a%10) {
				return true;
			}
		}
		else if(a <= 99 && a >= 10) {
			if(a/10 <= a % 10) {
				return true;
			}
		}
		else if(10 > a && a > 0) {
			return true;
		}
		return false;
	}
}
